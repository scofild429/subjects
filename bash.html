17-Professional Menus






#! /bin/bash
select car in BMW ROVER AUDI BENLI
do
    case $car in
        BMW)
            echo "BMW SELECTED";;
        ROVER)
            echo "ROVER SELECTED";;
        AUDI)
            echo "AUDI SELECTED";;
        BENLI)
            echo "BENLI SELECTED";;
        *)
            echo "ERROR! please only give number 0 to 4";;
        esac
done
